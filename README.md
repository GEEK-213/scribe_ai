## âš™ï¸ Installation & Setup

### 1. Database Setup (Supabase)
1.  Go to your Supabase Dashboard -> **SQL Editor**.
2.  Run the following SQL commands to create the necessary tables:

```sql
-- Create Tables
create table public.folders (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  name text not null,
  user_id uuid references auth.users not null
);

create table public.notes (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  title text,
  audio_path text,
  transcript text,
  summary text,
  quiz jsonb,
  status text default 'Processing',
  folder_id bigint references public.folders(id),
  user_id uuid references auth.users not null
);

create table public.study_tasks (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  title text not null,
  due_date text, 
  is_completed boolean default false,
  origin_note_id bigint references public.notes(id) on delete cascade,
  user_id uuid references auth.users not null
);

create table public.flashcards (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  front text not null,
  back text not null,
  note_id bigint references public.notes(id) on delete cascade
);

create table public.chat_messages (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  question text not null,
  response text,
  note_id bigint references public.notes(id) on delete cascade,
  user_id uuid references auth.users not null
);

-- Enable Realtime
alter publication supabase_realtime add table folders, notes, study_tasks, flashcards, chat_messages;

-- Create Storage Bucket
insert into storage.buckets (id, name, public) values ('Lectures', 'Lectures', true);
